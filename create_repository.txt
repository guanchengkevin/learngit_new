// 先创建一个目录，切换到目录 
mkdir learngit; 
cd learngit;

//初始化为仓库 repository
git init; 

//添加文件
git add file1 file2 ; 
//将文件提交到仓库
git commit -m "instructions"

//查看当前仓库的状态的变化(新增或者修改) 
git status; 
//查看当前仓库的日志
git log;

//git reflog 记录了所有的命令

cat file_name 
//可以查看repository中

//对工作区的文件进行回退,顺序：先暂存区，再仓库(实质是用暂存区或者仓库里面的文件替换工作区)
git checkout -- file_name;

//回退版本库
git reset;
